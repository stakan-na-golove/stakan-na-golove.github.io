┌──────────────────┐──┐
└──────────────────┘──┘                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│                     renderMap(geoData)                           │
│                                                                  │
│   1. Создаёт SVG элемент с D3.js                                │
│   2. Для каждого feature (региона) из GeoJSON:                  │
│      - Рисует path (границы региона)                            │
│      - Вызывает getRegionData(name) для привязки                │
│      - Вызывает getColor(value) для цвета                       │
│      - Добавляет обработчики hover/click                        │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│              getRegionData(geoJsonName) → Region                 │
│                                                                  │
│   Входные данные:                                                │
│   - geoJsonName = "Омская область" или "RU-OMS"                 │
│                                                                  │
│   Процесс поиска:                                               │
│   1. Ищет в isoToRegion['RU-OMS'] → "Омская область"           │
│   2. Ищет в exactGeoJsonMapping['омская область']               │
│   3. Ищет в REGIONS_DATA по совпадению имени                    │
│                                                                  │
│   Выход:                                                         │
│   - { id: 54, name: "Омская область", district: "Сибирский" }   │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│                         REGIONS_DATA                             │
│                                                                  │
│   Массив из 89 регионов:                                        │
│   [                                                              │
│     { id: 1, name: "Белгородская область", district: "ЦФО" },   │
│     { id: 2, name: "Брянская область", district: "ЦФО" },       │
│     ...                                                          │
│     { id: 54, name: "Омская область", district: "СФО" },        │
│     { id: 63, name: "Ханты-Мансийский АО — Югра", ... },        │
│     ...                                                          │
│   ]                                                              │
│                                                                  │
│   Используется для:                                              │
│   - Генерации таблицы                                           │
│   - Привязки данных к карте                                     │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│                      data (React State)                          │
│                                                                  │
│   const [data, setData] = useState<Record<number, number>>({})  │
│                                                                  │
│   Структура:                                                     │
│   {                                                              │
│     1: 45,      ← Белгородская область = 45                     │
│     54: 100,    ← Омская область = 100                          │
│     63: 75,     ← Ханты-Мансийский АО = 75                      │
│   }                                                              │
│                                                                  │
│   Ключ = id региона из REGIONS_DATA                             │
│   Значение = число, введённое пользователем                     │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│                    dataRef (React Ref)                           │
│                                                                  │
│   const dataRef = useRef(data)                                  │
│                                                                  │
│   Зачем нужен:                                                   │
│   - D3 события (hover, click) не видят актуальный state         │
│   - useRef хранит ссылку на актуальные данные                   │
│   - Обновляется через useEffect при изменении data              │
│                                                                  │
│   useEffect(() => {                                              │
│     dataRef.current = data                                       │
│   }, [data])                                                     │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│                   getColor(value) → цвет                         │
│                                                                  │
│   Входные данные:                                                │
│   - value = число из data[regionId]                             │
│   - minColor, midColor, maxColor — цвета градиента              │
│   - useMidColor — включена ли средняя точка                     │
│                                                                  │
│   Процесс:                                                       │
│   1. Находит min/max среди всех значений                        │
│   2. Нормализует value в диапазон 0-1                           │
│   3. Интерполирует цвет между min/mid/max                       │
│                                                                  │
│   Выход:                                                         │
│   - "#ff0000" (красный) или "#00ff00" (зелёный) и т.д.          │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│                      SVG на странице                             │
│                                                                  │
│   <svg ref={svgRef}>                                            │
│     <g class="map-group">                                        │
│       <path d="..." fill="#ff5555" />  ← Регион 1               │
│       <path d="..." fill="#55ff55" />  ← Регион 2               │
│       ...                                                        │
│     </g>                                                         │
│   </svg>                                                         │
│                                                                  │
│   Каждый path:                                                   │
│   - d = координаты границ (из GeoJSON geometry)                 │
│   - fill = цвет (из getColor)                                   │
│   - data-region-id = id региона (для связи с таблицей)          │
